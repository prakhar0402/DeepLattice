## Using Anaconda

# create conda environment
    conda create -n lattice python=3 
# activate environment
    source activate lattice
# install jupyter
    conda install jupyter
# install libraries numpy scipy networkx
    conda install numpy scipy networkx
# install matplotlib
    conda install -c conda-forge matplotlib
# install scikit-image
    conda install scikit-image
# install trimesh and dependencies
    conda install -c conda-forge rtree shapely
    conda install -c conda-forge pyembree
    pip install trimesh[easy]
# install meshpy (something might be missing here.. lol)
    git clone --recursive http://git.tiker.net/trees/meshpy.git
    cd meshpy
    python ./configure.py
    python setup.py install
